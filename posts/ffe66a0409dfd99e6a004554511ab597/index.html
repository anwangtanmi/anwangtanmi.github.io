1） 读入该文件，统计并输出单词的个数

 

 

2） 返回单词列表（注意：指返回单词列表即可，不要在屏幕上显示）

import re
#统计单词个数：
def count_words(file_path):
    with open(file_path) as file:
        text = file.read()
        words = re.findall(r'[a-zA-Z]+', text)
        count = len(words)
        return count
print(count_words('C:/Users/uestc2020/Desktop/words.txt'))
